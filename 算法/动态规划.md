## 背包问题
### 0-1背包问题

背包问题是已知一个数组表示每个物品的重量，一个数组表示每个物品的价值，W表示背包所能承受的最大重量。问题是求背包所能装物品的最大价值。
0-1是表示，每一个物品的选择状态只有装和不装两种。

```js
/** 
 * 背包问题，求背包所能装下的最大价值
 * 滚动数组版本
 * @param {Array.<number>} w 输入所有物品的重量
 * @param {Array.<number>} v 输入所有物品的价值
 * @param {number} W 背包所能装下的最大重量
 * */ 

const knapsack = function(w, v, W){
    const matrix = [...Array(2).fill(Array(W + 1).fill(0))];
    let temp1 = void 0;
    let line1 = 0; let line2 = 1;
    for(let i = 0; i <= W; i++){
        for(let j = 1; j <= w.length; j++){
            temp1 = matrix[line1][i - w[j]] + v[j];
            matrix[line2][i] = w[j] <= i ? Math.max(matrix[line1][i], temp1) : matrix[line1][i];
            [line1,line2] = [line2, line1];
        }
    }
    return matrix[line2][W];
}

/** 
 * 背包问题，求背包所能装下的最大价值
 * 可返回物品版本
 * @param {Array.<number>} w 输入所有物品的重量
 * @param {Array.<number>} v 输入所有物品的价值
 * @param {number} W 背包所能装下的最大重量
 * */ 
const knapsack = function(w, v, W){
    const matrix = [];
    for(let c = 0; c <= w.length; c++){
        matrix.push([...Array(W+1).fill(0)])
    }
    const re = [];
    let i = 0;
    let j = 1;
    for(i = 0; i <= W; i++){
        for(j = 1; j <= w.length; j++){
            matrix[j][i] = w[j-1] <= i ? Math.max(matrix[j - 1][i], matrix[j - 1][i - w[j - 1]] + v[j - 1]) : matrix[j-1][i];
        }
    }
    --i; --j;
    console.log(i,j)
    while(j>0 && i >= 0){
        console.log('m1',matrix[j][i],'m2', matrix[j-1][i])
        if(matrix[j][i] > matrix[j-1][i]){
            re.unshift(j - 1);
            console.log('w',w[j-1]);
            i = i-w[j-1];
        }
        j--;
        console.log(i,j)
    }
    return [matrix[w.length][W], re, matrix];
}
```
